	\begin{itemize}
		\item Schnell hohe Geschwindigkeiten
		\item Lösungsansatz: Maximale Geschwindigkeit $V_{max,j}$
			\begin{equation*}
				v_{ij}(t+1) = 
				\begin{cases}
					v_{ij}(t+1) & \text{if} |v_{ij}(t+1)| < V_{max,j} \\
					V_{max,j} & \text{if} |v_{ij}(t+1)| \geq V_{max,j}
				\end{cases}
			\end{equation*} 
	\end{itemize}
